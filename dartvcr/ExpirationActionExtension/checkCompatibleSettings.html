void checkCompatibleSettings(ExpirationAction action, Mode mode) {
  if (action == ExpirationAction.recordAgain && mode == Mode.replay) {
    throw VCRException(
        "Cannot use the recordAgain expiration action in combination with replay mode.");
  }
}
